audit.py
import os, sys
FORBIDDEN = ['predict_proba', 'mock', 'dummy_data', 'synthetic', 'np.random.normal']
def run_audit():
    issues = []
    for root, _, files in os.walk("./phase2"):
        for file in files:
            if file.endswith(".py"):
                with open(os.path.join(root, file), 'r') as f:
                    if any(w in f.read().lower() for w in FORBIDDEN):
                        issues.append(f"FAILURE: {w} in {file}")
    if issues: print("\n".join(issues)); sys.exit(1)
    print("SUCCESS"); sys.exit(0)
if __name__ == "__main__": run_audit()
